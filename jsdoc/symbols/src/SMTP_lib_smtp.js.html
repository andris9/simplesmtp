 15 /**
 16  * <p>Constructs a SMTP server</p>
 17  * 
 18  * <p>Possible options are:</p>
 19  * 
 20  * <ul>
 21  *     <li><b>HOSTNAME</b> - the hostname of the server, will be used for
 22  *         informational messages</li>
 23  *     <li><b>debug</b> - if set to true, print out messages about the connection</li>
 24  *     <li><b>timeout</b> - client timeout in milliseconds, defaults to 60 000</li>
 25  *     <li><b>SMTPBanner</b> - greeting banner that is sent to the client on connection</li>
 26  *     <li><b>requireAuthentication</b> - if set to true, require that the client
 27  *         must authenticate itself</li>
 28  *     <li><b>validateSender</b> - if set to true, emit <code>'validateSender'</code>
 29  *         with <code>email</code> and <code>callback</code> when the client
 30  *         enters <code>MAIL FROM:<address></code></li>
 31  *     <li><b>validateRecipients</b> - if set to true, emit <code>'validateRecipient'</code>
 32  *         with <code>email</code> and <code>callback</code> when the client
 33  *         enters <code>RCPT TO:<address></code></li>
 34  *     <li><b>maxSize</b> - maximum size of an e-mail in bytes</li>
 35  *     <li><b>credentials</b> - TLS credentials</li>
 36  *     <li><b>authMethods</b> - allowed authentication methods, defaults to <code>["PLAIN", "LOGIN"]</code></li>
 37  *     <li><b>disableEHLO</b> - if set, support HELO only</li>
 38  * </ul>
 39  * 
 40  * @constructor
 41  * @param {Object} [options] Options object
 42  */
 43 function SMTPServer(options){
 44     EventEmitter.call(this);
 45     
 46     this.options = options || {};
 47     this.options.HOSTNAME = this.options.HOSTNAME || (oslib.hostname && oslib.hostname()) ||
 48                (oslib.getHostname && oslib.getHostname()) ||
 49                "127.0.0.1";
 50 
 51     this.options.authMethods = (this.options.authMethods || ["PLAIN", "LOGIN"]).map(
 52         function(auth){
 53             return auth.toUpperCase().trim();
 54         });
 55     
 56     this.options.disableEHLO = !!this.options.disableEHLO;
 57 
 58     this.SMTPServer = new RAIServer({
 59         timeout: this.options.timeout || 60*1000,
 60         disconnectOnTimeout: false,
 61         debug: !!this.options.debug
 62     });
 63     
 64     this.SMTPServer.on("connection", this._createSMTPConnection.bind(this));
 65 }
 66 utillib.inherits(SMTPServer, EventEmitter);

 98 /**
 99  * <p>Sets up a handler for the connected client</p>
100  * 
101  * <p>Restarts the state and sets up event listeners for client actions</p>
102  * 
103  * @constructor
104  * @param {Object} server {@link SMTPServer} instance
105  * @param {Object} client RAISocket instance for the client
106  */
107 function SMTPConnection(server, client){
108     this.server = server;
109     this.client = client;
110 
111     this.init();
112 
113     if(this.server.options.debug){
114         console.log("Connection from", this.client.remoteAddress);
115     }
116     
117     this.client.on("timeout", this._onTimeout.bind(this));
118     this.client.on("error", this._onError.bind(this));
119     this.client.on("command", this._onCommand.bind(this));
120     this.client.on("end", this._onEnd.bind(this));
121     
122     this.client.on("data", this._onData.bind(this));
123     this.client.on("ready", this._onDataReady.bind(this));
124     
125     // Send the greeting banner
126     this.client.send("220 "+this.server.options.HOSTNAME+" "+(this.server.options.SMTPBanner || "ESMTP node.js simplesmtp"));
127 }

135  * @param {Boolean} [keepAuthData=false] If set to true keep authentication data

137 SMTPConnection.prototype.init = function(keepAuthData){
138     this.envelope = {from: "", to:[], date: new Date()};
139     
140     if(this.hostnamehostNameAppearsAs){
141         this.envelope.host = this.hostNameAppearsAs;
142     }
143     
144     if(this.client.remoteAddress){
145         this.envelope.remoteAddress = this.client.remoteAddress;
146     }
147     
148     if(!keepAuthData){
149         this.authentication = {
150             username: false,
151             authenticated: false,
152             state: "NORMAL"
153         };
154     }
155 };

198 /**
199  * <p>Will be called when a command is received from the client</p>
200  * 
201  * <p>If there's curently an authentication process going on, route
202  * the data to <code>_handleAuthLogin</code>, otherwise act as
203  * defined</p>
204  * 
205  * @event
206  * @param {String} command Command
207  * @param {Buffer} command Payload related to the command
208  */
209 SMTPConnection.prototype._onCommand = function(command, payload){
210 
211     if(this.authentication.state == "AUTHENTICATING"){
212         this._handleAuthLogin(command);
213         return;
214     }
215     
216     switch(command){
217         
218         // Should not occur too often 
219         case "HELO":
220             this._onCommandHELO(payload.toString("utf-8").trim());
221             break;
222             
223         // Lists server capabilities
224         case "EHLO":
225             if(!this.server.options.disableEHLO){
226                 this._onCommandEHLO(payload.toString("utf-8").trim());
227             }else{
228                 this.client.send("502 5.5.2 Error: command not recognized");
229             }
230             break;
231             
232         // Closes the connection
233         case "QUIT":
234             this.end("221 2.0.0 Goodbye!");
235             break;
236 
237         // Resets the current state
238         case "RSET":
239             this._onCommandRSET();
240             break;
241         
242         // Doesn't work for spam related purposes
243         case "VRFY":
244             this.end("252 2.1.5 Send some mail, I'll try my best");
245             break;
246         
247         // Initiate an e-mail by defining a sender
248         case "MAIL":
249             this._onCommandMAIL(payload.toString("utf-8").trim());
250             break;
251         
252         // Add recipients to the e-mail envelope
253         case "RCPT":
254             this._onCommandRCPT(payload.toString("utf-8").trim());
255             break;
256         
257         // Authenticate if needed
258         case "AUTH":
259             this._onCommandAUTH(payload);
260             break;
261         
262         // Start accepting binary data stream
263         case "DATA":
264             this._onCommandDATA();
265             break;
266         
267         // Upgrade connection to secure TLS
268         case "STARTTLS":
269             this._onCommandSTARTTLS();
270             break;
271             
272         // Display an error on anything else
273         default:
274             this.client.send("502 5.5.2 Error: command not recognized");
275     }  
276 };

278 /**
279  * <p>Initiate an e-mail by defining a sender.</p>
280  * 
281  * <p>This doesn't work if authorization is required but the client is
282  * not logged in yet.</p>
283  * 
284  * <p>If <code>validateSender</code> option is set to true, then emits
285  * <code>'validateSender'</code> and wait for the callback before moving
286  * on</p>
287  * 
288  * @param {String} mail Address payload in the form of "FROM:<address>"
289  */
290 SMTPConnection.prototype._onCommandMAIL = function(mail){
291     var match, email, domain;
292     
293     if(!this.hostNameAppearsAs){
294         return this.client.send("503 5.5.1 Error: send HELO/EHLO first");
295     }
296     
297     if(this.server.options.requireAuthentication && !this.authentication.authenticated){
298         return this.client.send("530 5.5.1 Authentication Required");
299     }
300     
301     if(this.envelope.from){
302         return this.client.send("503 5.5.1 Error: nested MAIL command");
303     }
304     
305     if(!(match = mail.match(/^from\:\s*<([^@>]+\@([^@>]+))>$/i))){
306         return this.client.send("501 5.1.7 Bad sender address syntax");
307     }
308     
309     email = match[1] || "";
310     domain = (match[2] || "").toLowerCase();
311     
312     dnslib.resolveMx(domain, (function(err, addresses){
313         if(err || !addresses || !addresses.length){
314             return this.client.send("450 4.1.8 <"+email+">: Sender address rejected: Domain not found");
315         }
316         
317         if(this.server.options.validateSender){
318             this.server.emit("validateSender", email, (function(err){
319                 if(err){
320                     return this.client.send("550 5.1.1 <"+email+">: Sender address rejected: User unknown in local sender table");
321                 }
322                 
323                 // force domain part to be lowercase
324                 email = email.substr(0, email.length - domain.length) + domain;
325                 this.envelope.from = email;
326                 this.client.send("250 2.1.0 Ok");
327         
328             }).bind(this));
329         }else{
330             // force domain part to be lowercase
331             email = email.substr(0, email.length - domain.length) + domain;
332             this.envelope.from = email;
333             this.client.send("250 2.1.0 Ok");
334         }
335     }).bind(this));  
336 };

338 /**
339  * <p>Add recipients to the e-mail envelope</p>
340  * 
341  * <p>This doesn't work if <code>MAIL</code> command is not yet executed</p>
342  * 
343  * <p>If <code>validateRecipients</code> option is set to true, then emits
344  * <code>'validateRecipient'</code> and wait for the callback before moving
345  * on</p>
346  * 
347  * @param {String} mail Address payload in the form of "TO:<address>"
348  */
349 SMTPConnection.prototype._onCommandRCPT = function(mail){
350     var match, email, domain;
351     
352     if(!this.envelope.from){
353         return this.client.send("503 5.5.1 Error: need MAIL command");
354     }
355     
356     if(!(match = mail.match(/^to\:\s*<([^@>]+\@([^@>]+))>$/i))){
357         return this.client.send("501 5.1.7 Bad recipient address syntax");
358     }
359     
360     email = match[1] || "";
361     domain = (match[2] || "").toLowerCase();
362     
363     dnslib.resolveMx(domain, (function(err, addresses){
364         if(err || !addresses || !addresses.length){
365             return this.client.send("450 4.1.8 <"+email+">: Recipient address rejected: Domain not found");
366         }
367         
368         if(this.server.options.validateRecipients){
369             this.server.emit("validateRecipient", email, (function(err){
370                 if(err){
371                     return this.client.send("550 5.1.1 <"+email+">: Recipient address rejected: User unknown in local recipient table");
372                 }
373                 
374                 // force domain part to be lowercase
375                 email = email.substr(0, email.length - domain.length) + domain;
376                 
377                 // add to recipients list
378                 if(this.envelope.to.indexOf(email)<0){
379                     this.envelope.to.push(email);
380                 }
381                 this.client.send("250 2.1.0 Ok");
382             }).bind(this));
383         }else{
384             // force domain part to be lowercase
385             email = email.substr(0, email.length - domain.length) + domain;
386             
387             // add to recipients list
388             if(this.envelope.to.indexOf(email)<0){
389                 this.envelope.to.push(email);
390             }
391             this.client.send("250 2.1.0 Ok");
392         }
393     }).bind(this));  
394 };

396 /**
397  * <p>Switch to data mode and starts waiting for a binary data stream. Emits
398  * <code>'startData'</code>.</p>
399  * 
400  * <p>If <code>RCPT</code> is not yet run, stop</p>
401  */
402 SMTPConnection.prototype._onCommandDATA = function(){
403     
404     if(!this.envelope.to.length){
405         return this.client.send("503 5.5.1 Error: need RCPT command");
406     }
407     
408     this.client.startDataMode();
409     this.client.send("354 End data with <CR><LF>.<CR><LF>");
410     this.server.emit("startData", this.envelope);
411 };

421 /**
422  * <p>If the server is in secure connection mode, start the authentication
423  * process. Param <code>payload</code> defines the authentication mechanism.</p>
424  * 
425  * <p>Currently supported - PLAIN and LOGIN. There is no need for more
426  * complicated mechanisms (different CRAM versions etc.) since authentication
427  * is only done in secure connection mode</p>
428  * 
429  * @param {Buffer} payload Defines the authentication mechanism
430  */
431 SMTPConnection.prototype._onCommandAUTH = function(payload){
432     var method;
433     
434     if(!this.server.options.requireAuthentication){
435         return this.client.send("503 5.5.1 Error: authentication not enabled");
436     }
437     
438     if(!this.client.secureConnection){
439         return this.client.send("530 5.7.0 Must issue a STARTTLS command first");
440     }
441     
442     if(this.authentication.authenticated){
443         return this.client.send("503 5.7.0 No identity changes permitted");
444     }
445     
446     payload = payload.toString("utf-8").trim().split(" ");
447     method = payload.shift().trim().toUpperCase();
448     
449     if(this.server.options.authMethods.indexOf(method)<0){
450         return this.client.send("535 5.7.8 Error: authentication failed: no mechanism available");
451     }
452     
453     switch(method){
454         case "PLAIN":
455             this._handleAuthPlain(payload);
456             break;
457         case "LOGIN":
458             this._handleAuthLogin();
459             break;
460     }
461 };

463 /**
464  * <p>Upgrade the connection to a secure TLS connection</p>
465  */
466 SMTPConnection.prototype._onCommandSTARTTLS = function(){
467     if(this.client.secureConnection){
468         return this.client.send("554 5.5.1 Error: TLS already active");
469     }
470     
471     this.client.send("220 2.0.0 Ready to start TLS");
472     
473     this.client.startTLS(this.server.options.credentials, (function(){
474         // Connection secured
475         // nothing to do here, since it is the client that should 
476         // make the next move
477     }).bind(this));
478 };

480 /**
481  * <p>Retrieve hostname from the client. Not very important, since client
482  * IP is already known and the client can send fake data</p>
483  * 
484  * @param {String} host Hostname of the client
485  */
486 SMTPConnection.prototype._onCommandHELO = function(host){
487     if(!host){
488         return this.client.send("501 Syntax: EHLO hostname");
489     }else{
490         this.hostNameAppearsAs = host;
491         this.envelope.host = host;
492     }
493     this.client.send("250 "+this.server.options.HOSTNAME+" at your service, ["+
494         this.client.remoteAddress+"]");
495 };
496 
497 /**
498  * <p>Retrieve hostname from the client. Not very important, since client
499  * IP is already known and the client can send fake data</p>
500  * 
501  * <p>Additionally displays server capability list to the client</p>
502  * 
503  * @param {String} host Hostname of the client 
504  */
505 SMTPConnection.prototype._onCommandEHLO = function(host){
506     var response = [this.server.options.HOSTNAME+" at your service, ["+
507         this.client.remoteAddress+"]", "8BITMIME", "ENHANCEDSTATUSCODES"];
508     
509     if(this.server.options.maxSize){
510         response.push("SIZE "+this.server.options.maxSize);
511     }
512     
513     if(this.client.secureConnection && this.server.options.requireAuthentication){
514         response.push("AUTH "+this.server.options.authMethods.join(" "));
515         response.push("AUTH="+this.server.options.authMethods.join(" "));
516     }
517     
518     if(!this.client.secureConnection){
519         response.push("STARTTLS");
520     }
521     
522     if(!host){
523         return this.client.send("501 Syntax: EHLO hostname");
524     }else{
525         this.hostNameAppearsAs = host;
526         this.envelope.host = host;
527     }
528     
529     this.client.send(response.map(function(feature, i, arr){
530         return "250"+(i<arr.length-1?"-":" ")+feature;
531     }).join("\r\n"));    
532 };

534 /**
535  * <p>Detect login information from the payload and initiate authentication
536  * by emitting <code>'authorizeUser'</code> and waiting for its callback</p>
537  * 
538  * @param {Buffer} payload AUTH PLAIN login information
539  */
540 SMTPConnection.prototype._handleAuthPlain = function(payload){
541     var userdata = new Buffer(payload.join(" "), "base64"), password;
542     userdata = userdata.toString("utf-8").split("\u0000");
543     this.authentication.username = userdata[1] || userdata[0] || "";
544     password = userdata[2] || "";
545     
546     this.server.emit("authorizeUser", 
547         this.envelope, 
548         this.authentication.username, 
549         password, 
550         (function(err, success){
551             if(err || !success){
552                 this.authentication.authenticated = false;
553                 this.authentication.username = false;
554                 this.authentication.state = "NORMAL";
555                 return this.client.send("535 5.7.8 Error: authentication failed: generic failure");
556             }
557             this.client.send("235 2.7.0 Authentication successful");
558             this.authentication.authenticated = true;
559             this.authentication.state = "AUTHENTICATED";
560         }).bind(this));
561 };
562 
563 /**
564  * <p>Sets authorization state to "AUTHENTICATING" and reuqests for the
565  * username and password from the client</p>
566  * 
567  * <p>If username and password are set initiate authentication
568  * by emitting <code>'authorizeUser'</code> and waiting for its callback</p>
569  * 
570  * @param {Buffer} payload AUTH LOGIN login information
571  */
572 SMTPConnection.prototype._handleAuthLogin = function(payload){
573     if(this.authentication.state == "NORMAL"){
574         this.authentication.state = "AUTHENTICATING";
575         this.client.send("334 VXNlcm5hbWU6");
576     }else if(this.authentication.state == "AUTHENTICATING"){
577         if(this.authentication.username === false){
578             this.authentication.username = new Buffer(payload, "base64").toString("utf-8");
579             this.client.send("334 UGFzc3dvcmQ6");
580         }else{
581             this.authentication.state = "VERIFYING";
582             this.server.emit("authorizeUser", 
583                 this.envelope, 
584                 this.authentication.username, 
585                 new Buffer(payload, "base64").toString("utf-8"), 
586                 (function(err, success){
587                     if(err || !success){
588                         this.authentication.authenticated = false;
589                         this.authentication.username = false;
590                         this.authentication.state = "NORMAL";
591                         return this.client.send("535 5.7.8 Error: authentication failed: generic failure");
592                     }
593                     this.client.send("235 2.7.0 Authentication successful");
594                     this.authentication.authenticated = true;
595                     this.authentication.state = "AUTHENTICATED";
596                 }).bind(this));
597         }
598         
599     }
600 };

612 /**

616  * @event
617  */
618 SMTPConnection.prototype._onDataReady = function(){
619     this.server.emit("dataReady", this.envelope, (function(err, code){
620         this.init(true); //reset state, keep auth data
621         
622         if(err){
623             this.client.send("550 FAILED");
624         }else{
625             this.client.send("250 2.0.0 Ok: queued as "+(code || "FOOBARBAZ"));
626         }
627         
628     }).bind(this));
